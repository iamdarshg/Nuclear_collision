cmake_minimum_required(VERSION 3.15)
project(Hg199Simulation LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

file(GLOB SOURCES src/*.cpp)
file(GLOB HEADERS src/*.hpp)

add_executable(hg199_sim src/main.cpp)
target_include_directories(hg199_sim PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
